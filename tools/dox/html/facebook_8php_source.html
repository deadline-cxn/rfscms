00060   protected function initSharedSession() {
00061     $cookie_name = $this->getSharedSessionCookieName();
00062     if (isset($_COOKIE[$cookie_name])) {
00063       $data = $this->parseSignedRequest($_COOKIE[$cookie_name]);
00064       if ($data && !empty($data['domain']) &&
00065           self::isAllowedDomain($this->getHttpHost(), $data['domain'])) {
00066         // good case
00067         $this->sharedSessionID = $data['id'];
00068         return;
00069       }
00070       // ignoring potentially unreachable data
00071     }
00072     // evil/corrupt/missing case
00073     $base_domain = $this->getBaseDomain();
00074     $this->sharedSessionID = md5(uniqid(mt_rand(), true));
00075     $cookie_value = $this->makeSignedRequest(
00076       array(
00077         'domain' => $base_domain,
00078         'id' => $this->sharedSessionID,
00079       )
00080     );
00081     $_COOKIE[$cookie_name] = $cookie_value;
00082     if (!headers_sent()) {
00083       $expire = time() + self::FBSS_COOKIE_EXPIRE;
00084       setcookie($cookie_name, $cookie_value, $expire, '/', '.'.$base_domain);

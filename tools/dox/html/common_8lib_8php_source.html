01460 /**
01461  * Formats $value to the given length and appends SI prefixes
01462  * with a $length of 0 no truncation occurs, number is only formated
01463  * to the current locale
01464  *
01465  * examples:
01466  * <code>
01467  * echo PMA_formatNumber(123456789, 6);     // 123,457 k
01468  * echo PMA_formatNumber(-123456789, 4, 2); //    -123.46 M
01469  * echo PMA_formatNumber(-0.003, 6);        //      -3 m
01470  * echo PMA_formatNumber(0.003, 3, 3);      //       0.003
01471  * echo PMA_formatNumber(0.00003, 3, 2);    //       0.03 m
01472  * echo PMA_formatNumber(0, 6);             //       0
01473  * </code>
01474  *
01475  * @param double  $value          the value to format
01476  * @param integer $digits_left    number of digits left of the comma
01477  * @param integer $digits_right   number of digits right of the comma
01478  * @param boolean $only_down      do not reformat numbers below 1
01479  * @param boolean $noTrailingZero removes trailing zeros right of the comma
01480  *                                (default: true)
01481  *
01482  * @return  string   the formatted value and its unit
01483  *
01484  * @access  public
01485  */
01486 function PMA_formatNumber($value, $digits_left = 3, $digits_right = 0,
01487 $only_down = false, $noTrailingZero = true)
01488 {
01489     if ($value==0) {
01490         return '0';
01491     }
01492 
01493     $originalValue = $value;
01494     //number_format is not multibyte safe, str_replace is safe
01495     if ($digits_left === 0) {
01496         $value = number_format($value, $digits_right);
01497         if ($originalValue != 0 && floatval($value) == 0) {
01498             $value = ' <' . (1 / PMA_pow(10, $digits_right));
01499         }
01500 
01501         return PMA_localizeNumber($value);
01502     }
01503 
01504     // this units needs no translation, ISO
01505     $units = array(
01506         -8 => 'y',
01507         -7 => 'z',
01508         -6 => 'a',
01509         -5 => 'f',
01510         -4 => 'p',
01511         -3 => 'n',
01512         -2 => '&micro;',
01513         -1 => 'm',
01514         0 => ' ',
01515         1 => 'k',
01516         2 => 'M',
01517         3 => 'G',
01518         4 => 'T',
01519         5 => 'P',
01520         6 => 'E',
01521         7 => 'Z',
01522         8 => 'Y'
01523     );
01524 
01525     // check for negative value to retain sign
01526     if ($value < 0) {
01527         $sign = '-';
01528         $value = abs($value);
01529     } else {
01530         $sign = '';
01531     }
01532 
01533     $dh = PMA_pow(10, $digits_right);
01534 
01535     /*
01536      * This gives us the right SI prefix already,
01537      * but $digits_left parameter not incorporated
01538      */
01539     $d = floor(log10($value) / 3);
01540     /*
01541      * Lowering the SI prefix by 1 gives us an additional 3 zeros
01542      * So if we have 3,6,9,12.. free digits ($digits_left - $cur_digits)
01543      * to use, then lower the SI prefix
01544      */
01545     $cur_digits = floor(log10($value / PMA_pow(1000, $d, 'pow'))+1);
01546     if ($digits_left > $cur_digits) {
01547         $d-= floor(($digits_left - $cur_digits)/3);
01548     }
01549 
01550     if ($d<0 && $only_down) {
01551         $d=0;
01552     }
01553 
01554     $value = round($value / (PMA_pow(1000, $d, 'pow') / $dh)) /$dh;
01555     $unit = $units[$d];
01556 
01557     // If we dont want any zeros after the comma just add the thousand seperator
01558     if ($noTrailingZero) {
01559         $value = PMA_localizeNumber(
01560             preg_replace('/(?<=\d)(?=(\d{3})+(?!\d))/', ',', $value)
01561         );
01562     } else {
01563         //number_format is not multibyte safe, str_replace is safe
01564         $value = PMA_localizeNumber(number_format($value, $digits_right));
01565     }
01566 
01567     if ($originalValue!=0 && floatval($value) == 0) {
01568         return ' <' . (1 / PMA_pow(10, $digits_right)) . ' ' . $unit;
01569     }
01570 
01571     return $sign . $value . ' ' . $unit;
01572 } // end of the 'PMA_formatNumber' function

02292  *
02293  * @param int    $rows        Number of rows in the pagination set
02294  * @param int    $pageNow     current page number
02295  * @param int    $nbTotalPage number of total pages
02296  * @param int    $showAll     If the number of pages is lower than this
02297  *                            variable, no pages will be omitted in pagination
02298  * @param int    $sliceStart  How many rows at the beginning should always be shown?
02299  * @param int    $sliceEnd    How many rows at the end should always be shown?
02300  * @param int    $percent     Percentage of calculation page offsets to hop to a
02301  *                            next page
02302  * @param int    $range       Near the current page, how many pages should
02303  *                            be considered "nearby" and displayed as well?
02304  * @param string $prompt      The prompt to display (sometimes empty)
02305  *
02306  * @return string
02307  *
02308  * @access  public
02309  */
02310 function PMA_pageselector($rows, $pageNow = 1, $nbTotalPage = 1,
02311     $showAll = 200, $sliceStart = 5, $sliceEnd = 5, $percent = 20,
02312     $range = 10, $prompt = '')
02313 {
02314     $increment = floor($nbTotalPage / $percent);
02315     $pageNowMinusRange = ($pageNow - $range);
02316     $pageNowPlusRange = ($pageNow + $range);
02317 
02318     $gotopage = $prompt . ' <select id="pageselector" ';
02319     if ($GLOBALS['cfg']['AjaxEnable']) {
02320         $gotopage .= ' class="ajax"';
02321     }
02322     $gotopage .= ' name="pos" >' . "\n";
02323     if ($nbTotalPage < $showAll) {
02324         $pages = range(1, $nbTotalPage);
02325     } else {
02326         $pages = array();
02327 
02328         // Always show first X pages
02329         for ($i = 1; $i <= $sliceStart; $i++) {
02330             $pages[] = $i;
02331         }
02332 
02333         // Always show last X pages
02334         for ($i = $nbTotalPage - $sliceEnd; $i <= $nbTotalPage; $i++) {
02335             $pages[] = $i;
02336         }
02337 
02338         // Based on the number of results we add the specified
02339         // $percent percentage to each page number,
02340         // so that we have a representing page number every now and then to
02341         // immediately jump to specific pages.
02342         // As soon as we get near our currently chosen page ($pageNow -
02343         // $range), every page number will be shown.
02344         $i = $sliceStart;
02345         $x = $nbTotalPage - $sliceEnd;
02346         $met_boundary = false;
02347         while ($i <= $x) {
02348             if ($i >= $pageNowMinusRange && $i <= $pageNowPlusRange) {
02349                 // If our pageselector comes near the current page, we use 1
02350                 // counter increments
02351                 $i++;
02352                 $met_boundary = true;
02353             } else {
02354                 // We add the percentage increment to our current page to
02355                 // hop to the next one in range
02356                 $i += $increment;
02357 
02358                 // Make sure that we do not cross our boundaries.
02359                 if ($i > $pageNowMinusRange && ! $met_boundary) {
02360                     $i = $pageNowMinusRange;
02361                 }
02362             }
02363 
02364             if ($i > 0 && $i <= $x) {
02365                 $pages[] = $i;
02366             }
02367         }
02368 
02369 /*
02370     Add page numbers with "geometrically increasing" distances.
02371 
02372     This helps me a lot when navigating through giant tables.
02373 
02374     Test case: table with 2.28 million sets, 76190 pages. Page of interest is
02375     between 72376 and 76190.
02376     Selecting page 72376.
02377     Now, old version enumerated only +/- 10 pages around 72376 and the
02378     percentage increment produced steps of about 3000.
02379 
02380     The following code adds page numbers +/- 2,4,8,16,32,64,128,256 etc.
02381     around the current page.
02382 */
02383 
02384         $i = $pageNow;
02385         $dist = 1;
02386         while ($i < $x) {
02387             $dist = 2 * $dist;
02388             $i = $pageNow + $dist;
02389             if ($i > 0 && $i <= $x) {
02390                 $pages[] = $i;
02391             }
02392         }
02393 
02394         $i = $pageNow;
02395         $dist = 1;
02396         while ($i >0) {
02397             $dist = 2 * $dist;
02398             $i = $pageNow - $dist;
02399             if ($i > 0 && $i <= $x) {
02400                 $pages[] = $i;
02401             }
02402         }
02403 
02404         // Since because of ellipsing of the current page some numbers may be double,
02405         // we unify our array:
02406         sort($pages);
02407         $pages = array_unique($pages);
02408     }
02409 
02410     foreach ($pages as $i) {
02411         if ($i == $pageNow) {
02412             $selected = 'selected="selected" style="font-weight: bold"';
02413         } else {
02414             $selected = '';
02415         }
02416         $gotopage .= '                <option ' . $selected
02417             . ' value="' . (($i - 1) * $rows) . '">' . $i . '</option>' . "\n";
02418     }
02419 
02420     $gotopage .= ' </select><noscript><input type="submit" value="'
02421         . __('Go') . '" /></noscript>';
02422 
02423     return $gotopage;
02424 } // end function

02906 /**
02907  * Extracts the various parts from a field type spec
02908  *
02909  * @param string $fieldspec Field specification
02910  *
02911  * @return  array associative array containing type, spec_in_brackets
02912  *          and possibly enum_set_values (another array)
02913  */
02914 function PMA_extractFieldSpec($fieldspec)
02915 {
02916     $first_bracket_pos = strpos($fieldspec, '(');
02917     if ($first_bracket_pos) {
02918         $spec_in_brackets = chop(
02919             substr(
02920                 $fieldspec,
02921                 $first_bracket_pos + 1,
02922                 (strrpos($fieldspec, ')') - $first_bracket_pos - 1)
02923             )
02924         );
02925         // convert to lowercase just to be sure
02926         $type = strtolower(chop(substr($fieldspec, 0, $first_bracket_pos)));
02927     } else {
02928         $type = strtolower($fieldspec);
02929         $spec_in_brackets = '';
02930     }
02931 
02932     if ('enum' == $type || 'set' == $type) {
02933         // Define our working vars
02934         $enum_set_values = array();
02935         $working = "";
02936         $in_string = false;
02937         $index = 0;
02938 
02939         // While there is another character to process
02940         while (isset($fieldspec[$index])) {
02941             // Grab the char to look at
02942             $char = $fieldspec[$index];
02943 
02944             // If it is a single quote, needs to be handled specially
02945             if ($char == "'") {
02946                 // If we are not currently in a string, begin one
02947                 if (! $in_string) {
02948                     $in_string = true;
02949                     $working = "";
02950                 } else {
02951                     // Otherwise, it may be either an end of a string,
02952                     // or a 'double quote' which can be handled as-is
02953                     // Check out the next character (if possible)
02954                     $has_next = isset($fieldspec[$index + 1]);
02955                     $next = $has_next ? $fieldspec[$index + 1] : null;
02956 
02957                     //If we have reached the end of our 'working' string (because
02958                     //there are no more chars,or the next char is not another quote)
02959                     if (! $has_next || $next != "'") {
02960                         $enum_set_values[] = $working;
02961                         $in_string = false;
02962 
02963                     } elseif ($next == "'") {
02964                         // Otherwise, this is a 'double quote',
02965                         // and can be added to the working string
02966                         $working .= "'";
02967                         // Skip the next char; we already know what it is
02968                         $index++;
02969                     }
02970                 }
02971             } elseif ('\\' == $char
02972                 && isset($fieldspec[$index + 1])
02973                 && "'" == $fieldspec[$index + 1]
02974             ) {
02975                 // escaping of a quote?
02976                 $working .= "'";
02977                 $index++;
02978             } else {
02979                 // Otherwise, add it to our working string like normal
02980                 $working .= $char;
02981             }
02982             // Increment character index
02983             $index++;
02984         } // end while
02985         $printtype = $type . '(' .  str_replace("','", "', '", $spec_in_brackets) . ')';
02986         $binary = false;
02987         $unsigned = false;
02988         $zerofill = false;
02989     } else {
02990         $enum_set_values = array();
02991 
02992         /* Create printable type name */
02993         $printtype = strtolower($fieldspec);
02994 
02995         // Strip the "BINARY" attribute, except if we find "BINARY(" because
02996         // this would be a BINARY or VARBINARY field type;
02997         // by the way, a BLOB should not show the BINARY attribute
02998         // because this is not accepted in MySQL syntax.
02999         if (preg_match('@binary@', $printtype) && ! preg_match('@binary[\(]@', $printtype)) {
03000             $printtype = preg_replace('@binary@', '', $printtype);
03001             $binary = true;
03002         } else {
03003             $binary = false;
03004         }
03005         $printtype = preg_replace('@zerofill@', '', $printtype, -1, $zerofill_cnt);
03006         $zerofill = ($zerofill_cnt > 0);
03007         $printtype = preg_replace('@unsigned@', '', $printtype, -1, $unsigned_cnt);
03008         $unsigned = ($unsigned_cnt > 0);
03009         $printtype = trim($printtype);
03010 
03011     }
03012 
03013     $attribute     = ' ';
03014     if ($binary) {
03015         $attribute = 'BINARY';
03016     }
03017     if ($unsigned) {
03018         $attribute = 'UNSIGNED';
03019     }
03020     if ($zerofill) {
03021         $attribute = 'UNSIGNED ZEROFILL';
03022     }
03023 
03024     return array(
03025         'type' => $type,
03026         'spec_in_brackets' => $spec_in_brackets,
03027         'enum_set_values'  => $enum_set_values,
03028         'print_type' => $printtype,
03029         'binary' => $binary,
03030         'unsigned' => $unsigned,
03031         'zerofill' => $zerofill,
03032         'attribute' => $attribute,
03033     );
03034 }

03699 /**
03700  * Checks if the current user has a specific privilege and returns true if the
03701  * user indeed has that privilege or false if (s)he doesn't. This function must
03702  * only be used for features that are available since MySQL 5, because it
03703  * relies on the INFORMATION_SCHEMA database to be present.
03704  *
03705  * Example:   PMA_currentUserHasPrivilege('CREATE ROUTINE', 'mydb');
03706  *            // Checks if the currently logged in user has the global
03707  *            // 'CREATE ROUTINE' privilege or, if not, checks if the
03708  *            // user has this privilege on database 'mydb'.
03709  *
03710  * @param string $priv The privilege to check
03711  * @param mixed  $db   null, to only check global privileges
03712  *                     string, db name where to also check for privileges
03713  * @param mixed  $tbl  null, to only check global/db privileges
03714  *                     string, table name where to also check for privileges
03715  *
03716  * @return bool
03717  */
03718 function PMA_currentUserHasPrivilege($priv, $db = null, $tbl = null)
03719 {
03720     // Get the username for the current user in the format
03721     // required to use in the information schema database.
03722     $user = PMA_DBI_fetch_value("SELECT CURRENT_USER();");
03723     if ($user === false) {
03724         return false;
03725     }
03726     $user = explode('@', $user);
03727     $username  = "''";
03728     $username .= str_replace("'", "''", $user[0]);
03729     $username .= "''@''";
03730     $username .= str_replace("'", "''", $user[1]);
03731     $username .= "''";
03732     // Prepage the query
03733     $query = "SELECT `PRIVILEGE_TYPE` FROM `INFORMATION_SCHEMA`.`%s` "
03734            . "WHERE GRANTEE='%s' AND PRIVILEGE_TYPE='%s'";
03735     // Check global privileges first.
03736     if (PMA_DBI_fetch_value(
03737         sprintf(
03738             $query,
03739             'USER_PRIVILEGES',
03740             $username,
03741             $priv
03742         )
03743     )
03744     ) {
03745         return true;
03746     }
03747     // If a database name was provided and user does not have the
03748     // required global privilege, try database-wise permissions.
03749     if ($db !== null) {
03750         // need to escape wildcards in db and table names, see bug #3518484
03751         $db = str_replace(array('%', '_'), array('\%', '\_'), $db);
03752         $query .= " AND TABLE_SCHEMA='%s'";
03753         if (PMA_DBI_fetch_value(
03754             sprintf(
03755                 $query,
03756                 'SCHEMA_PRIVILEGES',
03757                 $username,
03758                 $priv,
03759                 PMA_sqlAddSlashes($db)
03760             )
03761         )
03762         ) {
03763             return true;
03764         }
03765     } else {
03766         // There was no database name provided and the user
03767         // does not have the correct global privilege.
03768         return false;
03769     }
03770     // If a table name was also provided and we still didn't
03771     // find any valid privileges, try table-wise privileges.
03772     if ($tbl !== null) {
03773         // need to escape wildcards in db and table names, see bug #3518484
03774         $tbl = str_replace(array('%', '_'), array('\%', '\_'), $tbl);
03775         $query .= " AND TABLE_NAME='%s'";
03776         if ($retval = PMA_DBI_fetch_value(
03777             sprintf(
03778                 $query,
03779                 'TABLE_PRIVILEGES',
03780                 $username,
03781                 $priv,
03782                 PMA_sqlAddSlashes($db),
03783                 PMA_sqlAddSlashes($tbl)
03784             )
03785         )
03786         ) {
03787             return true;
03788         }
03789     }
03790     // If we reached this point, the user does not
03791     // have even valid table-wise privileges.
03792     return false;
03793 }

03851 function PMA_parseEnumSetValues($definition)
03852 {
03853     $values_string = htmlentities($definition);
03854     // There is a JS port of the below parser in functions.js
03855     // If you are fixing something here,
03856     // you need to also update the JS port.
03857     $values = array();
03858     $in_string = false;
03859     $buffer = '';
03860     for ($i=0; $i<strlen($values_string); $i++) {
03861         $curr = $values_string[$i];
03862         $next = $i == strlen($values_string)-1 ? '' : $values_string[$i+1];
03863         if (! $in_string && $curr == "'") {
03864             $in_string = true;
03865         } else if ($in_string && $curr == "\\" && $next == "\\") {
03866             $buffer .= "&#92;";
03867             $i++;
03868         } else if ($in_string && $next == "'" && ($curr == "'" || $curr == "\\")) {
03869             $buffer .= "&#39;";
03870             $i++;
03871         } else if ($in_string && $curr == "'") {
03872             $in_string = false;
03873             $values[] = $buffer;
03874             $buffer = '';
03875         } else if ($in_string) {
03876              $buffer .= $curr;
03877         }
03878     }
03879     if (strlen($buffer) > 0) {
03880         // The leftovers in the buffer are the last value (if any)
03881         $values[] = $buffer;
03882     }

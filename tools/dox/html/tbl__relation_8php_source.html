00076 /**
00077  * Split a string on backquote pairs
00078  *
00079  * @param string $text original string
00080  *
00081  * @return  array   containing the elements (and their surrounding backquotes)
00082  *
00083  * @access  public
00084  */
00085 function PMA_backquote_split($text)
00086 {
00087     $elements = array();
00088     $final_pos = strlen($text) - 1;
00089     $pos = 0;
00090     while ($pos <= $final_pos) {
00091         $first_backquote = strpos($text, '`', $pos);
00092         $second_backquote = strpos($text, '`', $first_backquote + 1);
00093         // after the second one, there might be another one which means
00094         // this is an escaped backquote
00095         if ($second_backquote < $final_pos && '`' == $text[$second_backquote + 1]) {
00096             $second_backquote = strpos($text, '`', $second_backquote + 2);
00097         }
00098         if (false === $first_backquote || false === $second_backquote) {
00099             break;
00100         }
00101         $elements[] = substr($text, $first_backquote, $second_backquote - $first_backquote + 1);
00102         $pos = $second_backquote + 1;
00103     }
00104     return($elements);
00105 }

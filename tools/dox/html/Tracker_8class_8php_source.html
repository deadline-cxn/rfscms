00959         if (!PMA_DRIZZLE) {
00960             return $tracking_set;

00962 
00963         // init conversion array (key 3 doesn't exist in calculated array)

00967             $array = array();

00969                 $flag = 1 << $i;
00970                 $array[$flag] = $set[$i];
00971                 $array[$set[$i]] = $flag;
00972             }
00973             self::$tracking_set_flags = $array;

00975

00977             // int > string conversion
00978             $aflags = array();
00979             // count/2 - conversion table has both int > string and string > int values
00980             for ($i = 0; $i < count(self::$tracking_set_flags)/2; $i++) {
00981                 $flag = 1 << $i;
00982                 if ($tracking_set & $flag) {
00983                     $aflags[] = self::$tracking_set_flags[$flag];
00984                 }
00985             }
00986             $flags = implode(',', $aflags);
00987         } else {
00988             // string > int conversion
00989             $flags = 0;
00990             foreach (explode(',', $tracking_set) as $strflag) {
00991                 if ($strflag == '') {
00992                     continue;
00993                 }
00994                 $flags |= self::$tracking_set_flags[$strflag];
00995             }

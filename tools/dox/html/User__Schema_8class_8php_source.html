00658     /**
00659      * process internal and foreign key relations
00660      *
00661      * @param string  $db          The database name
00662      * @param integer $pageNumber  document number/Id
00663      * @param array   $cfgRelation relation settings
00664      *
00665      * @return void
00666      * @access private
00667      */
00668     public function processRelations($db, $pageNumber, $cfgRelation)
00669     {
00670         /*
00671          * A u t o m a t i c    l a y o u t
00672          *
00673          * There are 2 kinds of relations in PMA
00674          * 1) Internal Relations 2) Foreign Key Relations
00675          */
00676         if (isset($this->autoLayoutInternal) || isset($this->autoLayoutForeign)) {
00677             $all_tables = array();
00678         }
00679 
00680         if (isset($this->autoLayoutForeign)) {
00681             /*
00682              * get the tables list
00683              * who support FOREIGN KEY, it's not
00684              * important that we group together InnoDB tables
00685              * and PBXT tables, as this logic is just to put
00686              * the tables on the layout, not to determine relations
00687              */
00688             $tables = PMA_DBI_get_tables_full($db);
00689             $foreignkey_tables = array();
00690             foreach ($tables as $table_name => $table_properties) {
00691                 if (PMA_foreignkey_supported($table_properties['ENGINE'])) {
00692                     $foreignkey_tables[] = $table_name;
00693                 }
00694             }
00695             $all_tables = $foreignkey_tables;
00696             /*
00697              * could be improved by finding the tables which have the
00698              * most references keys and placing them at the beginning
00699              * of the array (so that they are all center of schema)
00700              */
00701             unset($tables, $foreignkey_tables);
00702         }
00703 
00704         if (isset($this->autoLayoutInternal)) {
00705             /*
00706              * get the tables list who support Internal Relations;
00707              * This type of relations will be created when
00708              * you setup the PMA tables correctly
00709              */
00710             $master_tables = 'SELECT COUNT(master_table), master_table'
00711                 . ' FROM ' . PMA_backquote($GLOBALS['cfgRelation']['db']) . '.'
00712                 . PMA_backquote($cfgRelation['relation'])
00713                 . ' WHERE master_db = \'' . PMA_sqlAddSlashes($db) . '\''
00714                 . ' GROUP BY master_table'
00715                 . ' ORDER BY COUNT(master_table) DESC';
00716             $master_tables_rs = PMA_query_as_controluser(
00717                 $master_tables, false, PMA_DBI_QUERY_STORE
00718             );
00719             if ($master_tables_rs && PMA_DBI_num_rows($master_tables_rs) > 0) {
00720                 /* first put all the master tables at beginning
00721                  * of the list, so they are near the center of
00722                  * the schema
00723                  */
00724                 while (list(, $master_table) = PMA_DBI_fetch_row($master_tables_rs)) {
00725                        $all_tables[] = $master_table;
00726                 }
00727 
00728                 /* Now for each master, add its foreigns into an array
00729                  * of foreign tables, if not already there
00730                  * (a foreign might be foreign for more than
00731                  * one table, and might be a master itself)
00732                  */
00733 
00734                 $foreign_tables = array();
00735                 foreach ($all_tables as $master_table) {
00736                     $foreigners = PMA_getForeigners($db, $master_table);
00737                     foreach ($foreigners as $foreigner) {
00738                         if (! in_array($foreigner['foreign_table'], $foreign_tables)) {
00739                             $foreign_tables[] = $foreigner['foreign_table'];
00740                         }
00741                     }
00742                 }
00743 
00744                 /*
00745                  * Now merge the master and foreign arrays/tables
00746                  */
00747                 foreach ($foreign_tables as $foreign_table) {
00748                     if (! in_array($foreign_table, $all_tables)) {
00749                         $all_tables[] = $foreign_table;
00750                     }
00751                 }
00752             }
00753         }
00754 
00755         if (isset($this->autoLayoutInternal) || isset($this->autoLayoutForeign)) {
00756             $this->addRelationCoordinates(
00757                 $all_tables, $pageNumber, $db, $cfgRelation
00758             );
00759         }
00760 
00761         $this->chosenPage = $pageNumber;
00762     }

00764     /**
00765      * Add X and Y coordinates for a table
00766      *
00767      * @param array   $all_tables  A list of all tables involved
00768      * @param integer $pageNumber  document number/Id
00769      * @param string  $db          The database name
00770      * @param array   $cfgRelation relation settings
00771      *
00772      * @return void
00773      * @access private
00774      */
00775     public function addRelationCoordinates($all_tables, $pageNumber, $db, $cfgRelation)
00776     {
00777         /*
00778          * Now generate the coordinates for the schema
00779          * in a clockwise spiral and add to co-ordinates table
00780          */
00781         $pos_x = 300;
00782         $pos_y = 300;
00783         $delta = 110;
00784         $delta_mult = 1.10;
00785         $direction = "right";
00786         foreach ($all_tables as $current_table) {
00787             /*
00788             * save current table's coordinates
00789             */
00790             $insert_query = 'INSERT INTO '
00791                 . PMA_backquote($GLOBALS['cfgRelation']['db']) . '.'
00792                 . PMA_backquote($cfgRelation['table_coords']) . ' '
00793                 . '(db_name, table_name, pdf_page_number, x, y) '
00794                 . 'VALUES (\'' . PMA_sqlAddSlashes($db) . '\', \''
00795                 . PMA_sqlAddSlashes($current_table) . '\',' . $pageNumber
00796                 . ',' . $pos_x . ',' . $pos_y . ')';
00797             PMA_query_as_controluser($insert_query, false);
00798 
00799             /*
00800              * compute for the next table
00801              */
00802             switch ($direction) {
00803             case 'right':
00804                 $pos_x    += $delta;
00805                 $direction = "down";
00806                 $delta    *= $delta_mult;
00807                 break;
00808             case 'down':
00809                 $pos_y    += $delta;
00810                 $direction = "left";
00811                 $delta    *= $delta_mult;
00812                 break;
00813             case 'left':
00814                 $pos_x    -= $delta;
00815                 $direction = "up";
00816                 $delta    *= $delta_mult;
00817                 break;
00818             case 'up':
00819                 $pos_y    -= $delta;
00820                 $direction = "right";
00821                 $delta    *= $delta_mult;
00822                 break;
00823             }
00824         }
00825     }

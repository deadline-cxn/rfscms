00323 /**
00324  * Returns the "Excel" column name (i.e. 1 = "A", 26 = "Z", 27 = "AA", etc.)
00325  *
00326  * This functions uses recursion to build the Excel column name.
00327  *
00328  * The column number (1-26) is converted to the responding ASCII character (A-Z) and returned.
00329  *
00330  * If the column number is bigger than 26 (= num of letters in alfabet),
00331  * an extra character needs to be added. To find this extra character, the number is divided by 26
00332  * and this value is passed to another instance of the same function (hence recursion).
00333  * In that new instance the number is evaluated again, and if it is still bigger than 26, it is divided again
00334  * and passed to another instance of the same function. This continues until the number is smaller than 26.
00335  * Then the last called function returns the corresponding ASCII character to the function that called it.
00336  * Each time a called function ends an extra character is added to the column name.
00337  * When the first function is reached, the last character is addded and the complete column name is returned.
00338  *
00339  * @access  public
00340  *
00341  * @param int $num
00342  * @return  string The column's "Excel" name
00343  */

00345 {
00346     $A = 65; // ASCII value for capital "A"
00347     $col_name = "";
00348 
00349     if ($num > 26) {
00350         $div = (int)($num / 26);
00351         $remain = (int)($num % 26);
00352 
00353         // subtract 1 of divided value in case the modulus is 0,
00354         // this is necessary because A-Z has no 'zero'
00355         if ($remain == 0) {
00356             $div--;
00357         }
00358 
00359         // recursive function call
00360         $col_name = PMA_getColumnAlphaName($div);
00361         // use modulus as new column number
00362         $num = $remain;
00363     }
00364 
00365     if ($num == 0) {
00366         // use 'Z' if column number is 0,
00367         // this is necessary because A-Z has no 'zero'
00368         $col_name .= chr(($A + 26) - 1);
00369     } else {
00370         // convert column number to ASCII character
00371         $col_name .= chr(($A + $num) - 1);
00372     }
00373 
00374     return $col_name;

00707  *
00708  * @access  public
00709  *
00710  * @param int    $last_cumulative_type  Last cumulative column type  (VARCHAR or INT or BIGINT or DECIMAL or NONE)
00711  * @param string &$cell                 String representation of the cell for which a best-fit type is to be determined
00712  * @return  int                           The MySQL type representation (VARCHAR or INT or BIGINT or DECIMAL or NONE)

00714 function PMA_detectType($last_cumulative_type, &$cell)
00715 {
00716     /**
00717      * If numeric, determine if decimal, int or bigint
00718      * Else, we call it varchar for simplicity
00719      */
00720 
00721     if (! strcmp('NULL', $cell)) {
00722         if ($last_cumulative_type === null || $last_cumulative_type == NONE) {
00723             return NONE;
00724         } else {
00725             return $last_cumulative_type;
00726         }
00727     } elseif (is_numeric($cell)) {
00728         if ($cell == (string)(float)$cell && strpos($cell, ".") !== false && substr_count($cell, ".") == 1) {
00729             return DECIMAL;
00730         } else {
00731         if (abs($cell) > 2147483647) {
00732             return BIGINT;
00733         } else {
00734             return INT;
00735         }
00736         }
00737     } else {
00738         return VARCHAR;

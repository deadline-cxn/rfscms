01709 /**
01710  * returns details about the TRIGGERs for a specific table or database
01711  *
01712  * @param string $db        db name
01713  * @param string $table     table name
01714  * @param string $delimiter the delimiter to use (may be empty)
01715  *
01716  * @return array information about triggers (may be empty)
01717  */
01718 function PMA_DBI_get_triggers($db, $table = '', $delimiter = '//')
01719 {
01720     if (PMA_DRIZZLE) {
01721         // Drizzle doesn't support triggers
01722         return array();
01723     }
01724 
01725     $result = array();
01726     if (! $GLOBALS['cfg']['Server']['DisableIS']) {
01727         // Note: in http://dev.mysql.com/doc/refman/5.0/en/faqs-triggers.html
01728         // their example uses WHERE TRIGGER_SCHEMA='dbname' so let's use this
01729         // instead of WHERE EVENT_OBJECT_SCHEMA='dbname'
01730         $query = 'SELECT TRIGGER_SCHEMA, TRIGGER_NAME, EVENT_MANIPULATION'
01731             . ', EVENT_OBJECT_TABLE, ACTION_TIMING, ACTION_STATEMENT'
01732             . ', EVENT_OBJECT_SCHEMA, EVENT_OBJECT_TABLE, DEFINER'
01733             . ' FROM information_schema.TRIGGERS'
01734             . ' WHERE TRIGGER_SCHEMA= \'' . PMA_sqlAddSlashes($db) . '\'';
01735 
01736         if (! empty($table)) {
01737             $query .= " AND EVENT_OBJECT_TABLE = '" . PMA_sqlAddSlashes($table) . "';";
01738         }
01739     } else {
01740         $query = "SHOW TRIGGERS FROM " . PMA_backquote($db);
01741         if (! empty($table)) {
01742             $query .= " LIKE '" . PMA_sqlAddSlashes($table, true) . "';";
01743         }
01744     }
01745 
01746     if ($triggers = PMA_DBI_fetch_result($query)) {
01747         foreach ($triggers as $trigger) {
01748             if ($GLOBALS['cfg']['Server']['DisableIS']) {
01749                 $trigger['TRIGGER_NAME'] = $trigger['Trigger'];
01750                 $trigger['ACTION_TIMING'] = $trigger['Timing'];
01751                 $trigger['EVENT_MANIPULATION'] = $trigger['Event'];
01752                 $trigger['EVENT_OBJECT_TABLE'] = $trigger['Table'];
01753                 $trigger['ACTION_STATEMENT'] = $trigger['Statement'];
01754                 $trigger['DEFINER'] = $trigger['Definer'];
01755             }
01756             $one_result = array();
01757             $one_result['name'] = $trigger['TRIGGER_NAME'];
01758             $one_result['table'] = $trigger['EVENT_OBJECT_TABLE'];
01759             $one_result['action_timing'] = $trigger['ACTION_TIMING'];
01760             $one_result['event_manipulation'] = $trigger['EVENT_MANIPULATION'];
01761             $one_result['definition'] = $trigger['ACTION_STATEMENT'];
01762             $one_result['definer'] = $trigger['DEFINER'];
01763 
01764             // do not prepend the schema name; this way, importing the
01765             // definition into another schema will work
01766             $one_result['full_trigger_name'] = PMA_backquote($trigger['TRIGGER_NAME']);
01767             $one_result['drop'] = 'DROP TRIGGER IF EXISTS '
01768                 . $one_result['full_trigger_name'];
01769             $one_result['create'] = 'CREATE TRIGGER '
01770                 . $one_result['full_trigger_name'] . ' '
01771                 . $trigger['ACTION_TIMING']. ' '
01772                 . $trigger['EVENT_MANIPULATION']
01773                 . ' ON ' . PMA_backquote($trigger['EVENT_OBJECT_TABLE'])
01774                 . "\n" . ' FOR EACH ROW '
01775                 . $trigger['ACTION_STATEMENT'] . "\n" . $delimiter . "\n";
01776 
01777             $result[] = $one_result;
01778         }
01779     }
01780 
01781     // Sort results by name
01782     $name = array();
01783     foreach ($result as $key => $value) {
01784         $name[] = $value['name'];
01785     }
01786     array_multisort($name, SORT_ASC, $result);
01787 
01788     return($result);
01789 }
